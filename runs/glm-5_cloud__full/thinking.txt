    x: number;
    y: number;
}

type Direction = 'UP' | 'DOWN' | 'LEFT' | 'RIGHT';

// State
let snake: Point[] = [];
let food: Point = { x: 0, y: 0 };

let nextDirection: Direction = 'RIGHT';

// DOM

    if (ctx) {

    }
}

function initGame(ctx: CanvasRenderingContext2D): void {
    // Reset state
    snake = [{ x: 10, y: 10 }, { x: 9, y: 10 }, { x: 8, y: 10 }];
    currentDirection = 'RIGHT';
    nextDirection = 'RIGHT';
    score = 0;
    isGameOver = false;
    updateScore();

    placeFood();

    if (gameInterval) clearInterval(gameInterval);
    gameInterval = setInterval(() => gameLoop(ctx), GAME_SPEED);
}

function gameLoop(ctx: CanvasRenderingContext2D): void {
    if (isGameOver) return;

    update();
    draw(ctx);
}

function update(): void {
    currentDirection = nextDirection;

    const head = { ...snake[0] }; // Copy head

    switch (currentDirection) {
        case 'UP': head.y--; break;
        case 'DOWN': head.y++; break;
        case 'LEFT': head.x--; break;
        case 'RIGHT': head.x++; break;
    }

    // Check collisions
    if (checkCollision(head)) {
        isGameOver = true;
        // Draw game over logic here or in draw
        return;
    }

    snake.unshift(head); // Add new head

    if (head.x === food.x && head.y === food.y) {
        score++;
        updateScore();
        placeFood();
    } else {
        snake.pop(); // Remove tail if no food eaten
    }
}

function checkCollision(head: Point): boolean {
    // Wall collision
    if (head.x < 0 || head.x >= GRID_WIDTH || head.y < 0 || head.y >= GRID_HEIGHT) {
        return true;
    }
    // Self collision (check against current snake, head isn't added yet)
    for (const segment of snake) {
        if (segment.x === head.x && segment.y === head.y) {
            return true;
        }
    }
    return false;
}

function placeFood(): void {
    while (true) {
        const x = Math.floor(Math.random() * GRID_WIDTH);
        const y = Math.floor(Math.random() * GRID_HEIGHT);
        
        // Check if food is on snake
        let onSnake = false;
        for (const segment of snake) {
            if (segment.x === x && segment.y === y) {
                onSnake = true;
                break;
            }
        }
        
        if (!onSnake) {
            food = { x, y };
            break;
        }
    }
}

function draw(ctx: CanvasRenderingContext2D): void {
    // Clear canvas
    ctx.fillStyle = 'white';
    ctx.fillRect(0, 0, canvas!.width, canvas!.height);

    if (isGameOver) {
        ctx.fillStyle = 'red';
        ctx.font = '30px Arial';
        ctx.textAlign = 'center';
        ctx.fillText('Game Over', canvas!.width / 2, canvas!.height / 2);
        return;
    }

    // Draw Food

    ctx.fillRect(food.x * TILE_SIZE, food.y * TILE_SIZE, TILE_SIZE, TILE_SIZE);

    // Draw Snake
    ctx.fillStyle = 'green';
    for (const segment of snake) {
        ctx.fillRect(segment.x * TILE_SIZE, segment.y * TILE_SIZE, TILE_SIZE - 1, TILE_SIZE - 1);
    }
}

function updateScore(): void {
    if (scoreElement) {
        scoreElement.innerText = score.toString();
    }
}

            if (currentDirection !== 'RIGHT') nextDirection = 'LEFT';
            break;
        case 'ArrowRight':
            if (currentDirection !== 'LEFT') nextDirection = 'RIGHT';
            break;
    }
